        function renderSummary() {

            const summary = budgetData.summary;

            const categoryLabels = [];

            const categoryData = [];


            let calculatedTotalSebelumPajak = 0;

            Object.keys(budgetData).forEach(key => {

                if (key !== 'summary' && budgetData[key].items) {

                    const total = budgetData[key].items.reduce((sum, item) => sum + item.total, 0);

                    calculatedTotalSebelumPajak += total;

                    categoryLabels.push(budgetData[key].name.substring(0,25) + (budgetData[key].name.length > 25 ? '...' : ''));

                    categoryData.push(total);

                }

            });

            

            // Verifikasi apakah total kalkulasi sesuai dengan yang ada di summary object

            if (Math.abs(calculatedTotalSebelumPajak - summary.totalSebelumPajak) > 1) { // Toleransi perbedaan pembulatan kecil

                 console.warn("Peringatan: Ada perbedaan antara total biaya kategori yang dihitung (" + formatCurrency(calculatedTotalSebelumPajak) + ") dan yang tersimpan di ringkasan (" + formatCurrency(summary.totalSebelumPajak) + "). Periksa kembali data sumber.");

                 // Jika diperlukan, Anda bisa memutuskan untuk memperbarui summary.totalSebelumPajak di sini

                 // summary.totalSebelumPajak = calculatedTotalSebelumPajak; // Baris ini bisa diaktifkan jika kalkulasi ulang dianggap lebih akurat

            }



            let summaryHTML = `

                <h2 class="text-2xl font-semibold text-sky-800 mb-1">${summary.name}</h2>
